@Qualifier(@QUALIFIER@)
@Slf4j

     */
    @Autowired
    public @ENTITY@List(@Qualifier(@QUALIFIER@) IAService service) {
        super(service);
        super.entityClazz = @ENTITY@.class;
    }// end of Vaadin/@Route constructor

     * Apertura del dialogo per una entity esistente oppure nuova <br>
     * Sovrascritto <br>

        @ENTITY@Dialog dialog = appContext.getBean(@ENTITY@Dialog.class, service, entityClazz);
        dialog.open(entityBean, EAOperation.edit, this::save, this::delete);
    }// end of method
